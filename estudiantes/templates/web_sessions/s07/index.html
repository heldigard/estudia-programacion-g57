游닀 Listas: Python vs JavaScript

Python

# Crear lista
frutas = ["manzana", "banana"]

# Agregar al final
frutas.append("naranja")

# Longitud
len(frutas)

# Acceder elemento
frutas[0]

# Buscar
"manzana" in frutas

# Recorrer
for fruta in frutas:
    print(fruta)

JavaScript

// Crear array
const frutas = ["manzana", "banana"];

// Agregar al final
frutas.push("naranja");

// Longitud
frutas.length;

// Acceder elemento
frutas[0];

// Buscar
frutas.includes("manzana");

// Recorrer
for (const fruta of frutas) {
    console.log(fruta);
}
